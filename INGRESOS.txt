import { useState } from 'react';
import PageHeader from '../components/PageHeader';
import { Search, Plus, Download, Package, CheckCircle2, AlertCircle, Clock, Eye } from 'lucide-react';

const INGRESOS = [
  { id: 'ING-001', fecha: '2026-03-25', proveedor: 'Textiles ABC', referencia: 'TEL-2024', descripcion: 'Tela Jean Azul', cantidad: 500, unidad: 'metros', estado: 'recibido', responsable: 'Ana López' },
  { id: 'ING-002', fecha: '2026-03-25', proveedor: 'Hilos Norte', referencia: 'HIL-0891', descripcion: 'Hilo poliéster blanco', cantidad: 200, unidad: 'conos', estado: 'recibido', responsable: 'Carlos Ruiz' },
  { id: 'ING-003', fecha: '2026-03-24', proveedor: 'Botones & Cía', referencia: 'BOT-3344', descripcion: 'Botones nácar 12mm', cantidad: 5000, unidad: 'unidades', estado: 'parcial', responsable: 'María Pérez' },
  { id: 'ING-004', fecha: '2026-03-24', proveedor: 'Cremalleras SA', referencia: 'CRM-0021', descripcion: 'Cremallera YKK 20cm', cantidad: 1000, unidad: 'unidades', estado: 'pendiente', responsable: 'Pendiente' },
  { id: 'ING-005', fecha: '2026-03-23', proveedor: 'Textiles ABC', referencia: 'TEL-2025', descripcion: 'Tela lycra negra', cantidad: 300, unidad: 'metros', estado: 'recibido', responsable: 'José García' },
];

const estadoConfig = {
  recibido: { label: 'Recibido', color: 'bg-ok/10 text-ok border-ok/30', icon: CheckCircle2 },
  parcial: { label: 'Parcial', color: 'bg-warn/10 text-warn border-warn/30', icon: AlertCircle },
  pendiente: { label: 'Pendiente', color: 'bg-muted text-muted-foreground border-border', icon: Clock },
};

export default function Ingresos() {
  const [busqueda, setBusqueda] = useState('');
  const [filtroEstado, setFiltroEstado] = useState('todos');

  const filtrados = INGRESOS.filter(i => {
    const match = i.descripcion.toLowerCase().includes(busqueda.toLowerCase()) ||
      i.proveedor.toLowerCase().includes(busqueda.toLowerCase()) ||
      i.id.toLowerCase().includes(busqueda.toLowerCase());
    const matchE = filtroEstado === 'todos' || i.estado === filtroEstado;
    return match && matchE;
  });

  return (
    <div>
      <PageHeader
        title="Control de Ingresos"
        subtitle="Registro y seguimiento de ingresos de materiales"
        actions={
          <div className="flex gap-2">
            <button className="flex items-center gap-2 bg-card border border-border text-sm font-semibold px-3 py-2 rounded-lg hover:bg-muted transition-colors">
              <Download className="w-4 h-4" />
              Exportar
            </button>
            <button className="flex items-center gap-2 bg-primary text-white text-sm font-semibold px-4 py-2 rounded-lg hover:bg-primary/90 transition-colors shadow-sm">
              <Plus className="w-4 h-4" />
              Nuevo Ingreso
            </button>
          </div>
        }
      />

      {/* KPIs */}
      <div className="grid grid-cols-3 gap-3 mb-5">
        <div className="bg-ok/10 border border-ok/30 rounded-xl p-3 text-center">
          <p className="text-2xl font-barlow font-700 text-ok">{INGRESOS.filter(i => i.estado === 'recibido').length}</p>
          <p className="text-xs text-muted-foreground mt-0.5">Recibidos hoy</p>
        </div>
        <div className="bg-warn/10 border border-warn/30 rounded-xl p-3 text-center">
          <p className="text-2xl font-barlow font-700 text-warn">{INGRESOS.filter(i => i.estado === 'parcial').length}</p>
          <p className="text-xs text-muted-foreground mt-0.5">Parciales</p>
        </div>
        <div className="bg-muted border border-border rounded-xl p-3 text-center">
          <p className="text-2xl font-barlow font-700 text-muted-foreground">{INGRESOS.filter(i => i.estado === 'pendiente').length}</p>
          <p className="text-xs text-muted-foreground mt-0.5">Pendientes</p>
        </div>
      </div>

      {/* Filtros */}
      <div className="flex flex-col sm:flex-row gap-3 mb-5">
        <div className="relative flex-1">
          <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-muted-foreground" />
          <input
            type="text"
            placeholder="Buscar por proveedor, descripción o ID..."
            value={busqueda}
            onChange={e => setBusqueda(e.target.value)}
            className="w-full pl-9 pr-4 py-2.5 text-sm bg-card border border-border rounded-lg focus:outline-none focus:ring-2 focus:ring-primary/30 focus:border-primary"
          />
        </div>
        <div className="flex gap-2 flex-wrap">
          {['todos', 'recibido', 'parcial', 'pendiente'].map(e => (
            <button
              key={e}
              onClick={() => setFiltroEstado(e)}
              className={`px-3 py-2 rounded-lg text-sm font-semibold border capitalize transition-all ${
                filtroEstado === e
                  ? 'bg-primary text-white border-primary'
                  : 'bg-card text-muted-foreground border-border hover:border-primary/50'
              }`}
            >
              {e}
            </button>
          ))}
        </div>
      </div>

      {/* Tabla */}
      <div className="bg-card rounded-xl border border-border shadow-sm overflow-hidden">
        <div className="overflow-x-auto">
          <table className="w-full text-sm">
            <thead>
              <tr className="bg-muted/60 border-b border-border">
                <th className="text-left px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">ID</th>
                <th className="text-left px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Proveedor</th>
                <th className="text-left px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Material</th>
                <th className="text-center px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Cantidad</th>
                <th className="text-left px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Responsable</th>
                <th className="text-center px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Estado</th>
                <th className="text-center px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Fecha</th>
              </tr>
            </thead>
            <tbody>
              {filtrados.map((item, i) => {
                const cfg = estadoConfig[item.estado];
                const IconE = cfg.icon;
                return (
                  <tr key={i} className="border-b border-border last:border-0 hover:bg-muted/30 transition-colors">
                    <td className="px-4 py-3">
                      <span className="font-barlow font-700 text-primary text-base">{item.id}</span>
                    </td>
                    <td className="px-4 py-3 font-medium">{item.proveedor}</td>
                    <td className="px-4 py-3">
                      <p className="font-medium">{item.descripcion}</p>
                      <p className="text-xs text-muted-foreground">{item.referencia}</p>
                    </td>
                    <td className="px-4 py-3 text-center font-semibold">
                      {item.cantidad.toLocaleString()} <span className="text-xs text-muted-foreground font-normal">{item.unidad}</span>
                    </td>
                    <td className="px-4 py-3 text-sm text-muted-foreground">{item.responsable}</td>
                    <td className="px-4 py-3 text-center">
                      <span className={`inline-flex items-center gap-1.5 text-xs font-semibold px-2.5 py-1 rounded-full border ${cfg.color}`}>
                        <IconE className="w-3 h-3" />
                        {cfg.label}
                      </span>
                    </td>
                    <td className="px-4 py-3 text-center text-xs text-muted-foreground">{item.fecha}</td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>
        {filtrados.length === 0 && (
          <div className="text-center py-10 text-muted-foreground">
            <p className="text-sm">No se encontraron ingresos</p>
          </div>
        )}
      </div>
    </div>
  );
}
