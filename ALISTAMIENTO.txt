import { useState } from 'react';
import PageHeader from '../components/PageHeader';
import { Search, ScanLine, CheckCircle2, XCircle, Clock, AlertCircle, Filter } from 'lucide-react';

const ALISTAMIENTOS = [
  { ref: 'REF-1042', descripcion: 'Pantalón Jean Slim', talla: 'M', cantidad: 200, alistado: 200, estado: 'completo', operario: 'Juan Mora', hora: '07:15' },
  { ref: 'REF-1043', descripcion: 'Camisa Cuadros', talla: 'L', cantidad: 150, alistado: 98, estado: 'parcial', operario: 'Diana Cruz', hora: '08:00' },
  { ref: 'REF-1044', descripcion: 'Vestido Floral', talla: 'S', cantidad: 100, alistado: 0, estado: 'pendiente', operario: 'Sandra Gil', hora: '09:30' },
  { ref: 'REF-1045', descripcion: 'Chaqueta Deportiva', talla: 'XL', cantidad: 80, alistado: 80, estado: 'completo', operario: 'Luis Pardo', hora: '07:45' },
  { ref: 'REF-1046', descripcion: 'Blusa Escote V', talla: 'M', cantidad: 120, alistado: 45, estado: 'parcial', operario: 'Rosa Nieto', hora: '08:30' },
  { ref: 'REF-1047', descripcion: 'Short Deportivo', talla: 'S', cantidad: 60, alistado: 0, estado: 'pendiente', operario: 'Pendiente', hora: '—' },
];

const estadoConfig = {
  completo: { label: 'Completo', color: 'bg-ok/10 text-ok border-ok/30', icon: CheckCircle2 },
  parcial: { label: 'Parcial', color: 'bg-warn/10 text-warn border-warn/30', icon: AlertCircle },
  pendiente: { label: 'Pendiente', color: 'bg-muted text-muted-foreground border-border', icon: Clock },
};

export default function Alistamientos() {
  const [busqueda, setBusqueda] = useState('');
  const [filtroEstado, setFiltroEstado] = useState('todos');

  const filtrados = ALISTAMIENTOS.filter(a => {
    const matchBusqueda = a.ref.toLowerCase().includes(busqueda.toLowerCase()) ||
      a.descripcion.toLowerCase().includes(busqueda.toLowerCase());
    const matchEstado = filtroEstado === 'todos' || a.estado === filtroEstado;
    return matchBusqueda && matchEstado;
  });

  const totales = {
    completos: ALISTAMIENTOS.filter(a => a.estado === 'completo').length,
    parciales: ALISTAMIENTOS.filter(a => a.estado === 'parcial').length,
    pendientes: ALISTAMIENTOS.filter(a => a.estado === 'pendiente').length,
  };

  return (
    <div>
      <PageHeader
        title="Alistamientos"
        subtitle="Control de alistamiento de órdenes de producción"
        actions={
          <button className="flex items-center gap-2 bg-primary text-white text-sm font-semibold px-4 py-2 rounded-lg hover:bg-primary/90 transition-colors shadow-sm">
            <ScanLine className="w-4 h-4" />
            Escanear QR
          </button>
        }
      />

      {/* Resumen */}
      <div className="grid grid-cols-3 gap-3 mb-5">
        <div className="bg-ok/10 border border-ok/30 rounded-xl p-3 text-center">
          <p className="text-2xl font-barlow font-700 text-ok">{totales.completos}</p>
          <p className="text-xs text-muted-foreground mt-0.5">Completos</p>
        </div>
        <div className="bg-warn/10 border border-warn/30 rounded-xl p-3 text-center">
          <p className="text-2xl font-barlow font-700 text-warn">{totales.parciales}</p>
          <p className="text-xs text-muted-foreground mt-0.5">Parciales</p>
        </div>
        <div className="bg-muted border border-border rounded-xl p-3 text-center">
          <p className="text-2xl font-barlow font-700 text-muted-foreground">{totales.pendientes}</p>
          <p className="text-xs text-muted-foreground mt-0.5">Pendientes</p>
        </div>
      </div>

      {/* Buscador y filtros */}
      <div className="flex flex-col sm:flex-row gap-3 mb-5">
        <div className="relative flex-1">
          <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-muted-foreground" />
          <input
            type="text"
            placeholder="Buscar por referencia o descripción..."
            value={busqueda}
            onChange={e => setBusqueda(e.target.value)}
            className="w-full pl-9 pr-4 py-2.5 text-sm bg-card border border-border rounded-lg focus:outline-none focus:ring-2 focus:ring-primary/30 focus:border-primary"
          />
        </div>
        <div className="flex gap-2">
          {[
            { key: 'todos', label: 'Todos' },
            { key: 'completo', label: 'Completo' },
            { key: 'parcial', label: 'Parcial' },
            { key: 'pendiente', label: 'Pendiente' },
          ].map(({ key, label }) => (
            <button
              key={key}
              onClick={() => setFiltroEstado(key)}
              className={`px-3 py-2 rounded-lg text-sm font-semibold border transition-all ${
                filtroEstado === key
                  ? 'bg-primary text-white border-primary'
                  : 'bg-card text-muted-foreground border-border hover:border-primary/50'
              }`}
            >
              {label}
            </button>
          ))}
        </div>
      </div>

      {/* Tabla */}
      <div className="bg-card rounded-xl border border-border shadow-sm overflow-hidden">
        <div className="overflow-x-auto">
          <table className="w-full text-sm">
            <thead>
              <tr className="bg-muted/60 border-b border-border">
                <th className="text-left px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Referencia</th>
                <th className="text-left px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Descripción</th>
                <th className="text-center px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Talla</th>
                <th className="text-center px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Progreso</th>
                <th className="text-left px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Operario</th>
                <th className="text-center px-4 py-3 font-semibold text-muted-foreground text-xs uppercase tracking-wide">Estado</th>
              </tr>
            </thead>
            <tbody>
              {filtrados.map((a, i) => {
                const cfg = estadoConfig[a.estado];
                const IconE = cfg.icon;
                const porcentaje = Math.round((a.alistado / a.cantidad) * 100);
                return (
                  <tr key={i} className="border-b border-border last:border-0 hover:bg-muted/30 transition-colors">
                    <td className="px-4 py-3">
                      <span className="font-barlow font-700 text-primary text-base">{a.ref}</span>
                      <span className="block text-xs text-muted-foreground">{a.hora}</span>
                    </td>
                    <td className="px-4 py-3 font-medium">{a.descripcion}</td>
                    <td className="px-4 py-3 text-center">
                      <span className="inline-block bg-muted px-2.5 py-0.5 rounded-full text-xs font-semibold">{a.talla}</span>
                    </td>
                    <td className="px-4 py-3">
                      <div className="flex items-center gap-2">
                        <div className="flex-1 h-2 bg-muted rounded-full overflow-hidden min-w-[60px]">
                          <div
                            className={`h-full rounded-full ${a.estado === 'completo' ? 'bg-ok' : a.estado === 'parcial' ? 'bg-warn' : 'bg-muted-foreground/30'}`}
                            style={{ width: `${porcentaje}%` }}
                          />
                        </div>
                        <span className="text-xs font-semibold text-muted-foreground w-16 text-right">
                          {a.alistado}/{a.cantidad}
                        </span>
                      </div>
                    </td>
                    <td className="px-4 py-3 text-sm text-muted-foreground">{a.operario}</td>
                    <td className="px-4 py-3 text-center">
                      <span className={`inline-flex items-center gap-1.5 text-xs font-semibold px-2.5 py-1 rounded-full border ${cfg.color}`}>
                        <IconE className="w-3 h-3" />
                        {cfg.label}
                      </span>
                    </td>
                  </tr>
                );
              })}
            </tbody>
          </table>
        </div>
        {filtrados.length === 0 && (
          <div className="text-center py-10 text-muted-foreground">
            <p className="text-sm">No se encontraron alistamientos</p>
          </div>
        )}
      </div>
    </div>
  );
}
